-- Derzko UI Mobile Version
-- Адаптировано для мобильных устройств

repeat task.wait() until game:IsLoaded()
wait(0.5)

-- Проверяем, что это мобильное устройство
local isMobile = (game:GetService("UserInputService").TouchEnabled and not game:GetService("UserInputService").KeyboardEnabled)

-- Создаем основной интерфейс
local ScreenGui = Instance.new("ScreenGui")
ScreenGui.Name = "DerzkoUI"
ScreenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
ScreenGui.ResetOnSpawn = false

-- Адаптируем размеры под мобильные устройства
local containerWidth = isMobile and 300 or 400
local containerHeight = isMobile and 100 or 120
local fontSize = isMobile and 28 or 32

-- Создаем фоновый затемняющий слой
local Background = Instance.new("Frame")
Background.Name = "Background"
Background.Size = UDim2.new(1, 0, 1, 0)
Background.Position = UDim2.new(0, 0, 0, 0)
Background.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
Background.BackgroundTransparency = 0.7
Background.BorderSizePixel = 0
Background.Visible = true
Background.ZIndex = 1
Background.Parent = ScreenGui

-- Создаем основной контейнер для текста
local MainContainer = Instance.new("Frame")
MainContainer.Name = "MainContainer"
MainContainer.Size = UDim2.new(0, containerWidth, 0, containerHeight)
MainContainer.Position = UDim2.new(0.5, 0, 0.4, 0)
MainContainer.AnchorPoint = Vector2.new(0.5, 0.5)
MainContainer.BackgroundColor3 = Color3.fromRGB(25, 25, 25)
MainContainer.BackgroundTransparency = 0.2
MainContainer.BorderSizePixel = 0
MainContainer.Visible = true
MainContainer.ZIndex = 2
MainContainer.Parent = ScreenGui

-- Закругление углов контейнера
local ContainerCorner = Instance.new("UICorner")
ContainerCorner.CornerRadius = UDim.new(0.05, 0)
ContainerCorner.Parent = MainContainer

-- Заголовок Derzko UI
local Title = Instance.new("TextLabel")
Title.Name = "Title"
Title.Size = UDim2.new(0.9, 0, 0.7, 0)
Title.Position = UDim2.new(0.05, 0, 0.15, 0)
Title.BackgroundTransparency = 1
Title.Text = "Derzko UI"
Title.TextColor3 = Color3.fromRGB(255, 255, 255)
Title.TextScaled = false
Title.Font = Enum.Font.GothamBlack
Title.TextSize = fontSize
Title.TextXAlignment = Enum.TextXAlignment.Center
Title.Visible = true
Title.ZIndex = 3
Title.Parent = MainContainer

-- Подзаголовок
local Subtitle = Instance.new("TextLabel")
Subtitle.Name = "Subtitle"
Subtitle.Size = UDim2.new(0.9, 0, 0.3, 0)
Subtitle.Position = UDim2.new(0.05, 0, 0.7, 0)
Subtitle.BackgroundTransparency = 1
Subtitle.Text = "Your custom UI library"
Subtitle.TextColor3 = Color3.fromRGB(200, 200, 200)
Subtitle.TextScaled = false
Subtitle.Font = Enum.Font.Gotham
Subtitle.TextSize = fontSize - 8
Subtitle.TextXAlignment = Enum.TextXAlignment.Center
Subtitle.Visible = true
Subtitle.ZIndex = 3
Subtitle.Parent = MainContainer

-- Функция для анимации
local function playAnimation()
    -- Начальное состояние (невидимое)
    MainContainer.Position = UDim2.new(0.5, 0, 0.3, 0)
    MainContainer.Size = UDim2.new(0, 0, 0, 0)
    MainContainer.BackgroundTransparency = 1
    Title.TextTransparency = 1
    Subtitle.TextTransparency = 1
    
    -- Анимация появления
    local appearTween1 = game:GetService("TweenService"):Create(
        MainContainer,
        TweenInfo.new(0.8, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {Size = UDim2.new(0, containerWidth, 0, containerHeight), BackgroundTransparency = 0.2}
    )
    
    local appearTween2 = game:GetService("TweenService"):Create(
        MainContainer,
        TweenInfo.new(0.8, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {Position = UDim2.new(0.5, 0, 0.4, 0)}
    )
    
    local textAppearTween = game:GetService("TweenService"):Create(
        Title,
        TweenInfo.new(0.8, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {TextTransparency = 0}
    )
    
    local subtitleAppearTween = game:GetService("TweenService")::Create(
        Subtitle,
        TweenInfo.new(0.8, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {TextTransparency = 0}
    )
    
    appearTween1:Play()
    appearTween2:Play()
    textAppearTween:Play()
    subtitleAppearTween:Play()
    
    -- Ждем 0.8 секунды
    wait(0.8)
    
    -- Анимация исчезновения (уход вниз)
    local disappearTween1 = game:GetService("TweenService"):Create(
        MainContainer,
        TweenInfo.new(1.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {Position = UDim2.new(0.5, 0, 0.8, 0), BackgroundTransparency = 1}
    )
    
    local textDisappearTween = game:GetService("TweenService"):Create(
        Title,
        TweenInfo.new(1.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {TextTransparency = 1}
    )
    
    local subtitleDisappearTween = game:GetService("TweenService"):Create(
        Subtitle,
        TweenInfo.new(1.2, Enum.EasingStyle.Quint, Enum.EasingDirection.Out),
        {TextTransparency = 1}
    )
    
    disappearTween1:Play()
    textDisappearTween:Play()
    subtitleDisappearTween:Play()
    
    -- После завершения анимации удаляем GUI
    disappearTween1.Completed:Connect(function()
        ScreenGui:Destroy()
    end)
end

-- Функция для закрытия GUI
local function closeGUI()
    ScreenGui:Destroy()
end

-- Закрытие по клику (тапу)
Background.MouseButton1Click:Connect(closeGUI)
MainContainer.MouseButton1Click:Connect(closeGUI)

-- Запускаем анимацию
playAnimation()
